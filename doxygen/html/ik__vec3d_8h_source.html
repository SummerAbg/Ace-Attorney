203 void rotateXZBy(ik_f64 degrees, const vec3d<T> &center) {
204 degrees *= IK_DEGTORAD64;
205 T cs = (T)cos(degrees);
206 T sn = (T)sin(degrees);

210 X += center.X;
211 Z += center.Z;

265 vec3d<T> getHorizontalAngle() {
266 vec3d<T> angle;
267
268 angle.Y = (T)atan2(X, Z);
269 angle.Y *= (ik_f32)IK_RADTODEG;
270
271 if (angle.Y < 0.0f)
272 angle.Y += 360.0f;
273 if (angle.Y >= 360.0f)
274 angle.Y -= 360.0f;
275
276 ik_f32 z1 = (T)sqrt(X * X + Z * Z);
277
278 angle.X = (T)atan2(z1, Y);
279 angle.X *= (ik_f32)IK_RADTODEG;
280 angle.X -= 90.0f;
281
282 if (angle.X < 0.0f)
283 angle.X += 360.0f;
284 if (angle.X >= 360)
285 angle.X -= 360.0f;
286
287 return angle;
288 }
